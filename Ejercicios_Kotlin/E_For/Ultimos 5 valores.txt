//Desarrollar un programa que solicite la carga de 10 números e imprima la suma de los últimos 5 valores ingresados.



fun main(args: Array<String>) {



   var suma=0;



   for(i in 1..10 ){


       if(i>=5){

           suma=suma+i;

           println(suma)


       		}
    
	}

}
